import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner sc= new Scanner(System.in);
        int a= sc.nextInt();
        int max=0;
        while(a!=0){
            int b=a%10;
            if(b>max){
                max=b;
            }
            a=a/10;
        }
        System.out.println(max);
        
        
	}
}
