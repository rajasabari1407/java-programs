import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner sc= new Scanner(System.in);
        int a= sc.nextInt();
        int sum=0;
        while(a!=0){
            int b=a%10;
            a=a/10;
            sum=b+sum;
        }
        System.out.println(sum);
        
        
	}
}
